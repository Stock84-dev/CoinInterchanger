1. open cmd window in project folder
2. npm init 
3. npm install -S webpack
4. npm install -g webpack
5. create webpack.config.js and put this code https://gist.github.com/learncodeacademy/25092d8f1daf5e4a6fd3 // remember to change entry and output path
6. npm i -g webpack-cli // installing webpack-cli globally 
7. webpack // transpiling for production purposes or webpack --mode development
8. NODE_ENV=production webpack // transpiling for deployment
if loader are required then npm i babel-core babel-loader @babel/preset-env @babel/preset-react @babel/core -D
// NOTE: maybe all you need is @babel/name for every preset/env

// webpack.config.js
var debug = process.env.NODE_ENV !== "production";
var webpack = require('webpack');

module.exports = {
  context: __dirname,
  devtool: debug ? "inline-sourcemap" : null,
  entry: "./index.js",
  output: {
    path: __dirname + "/js",
    filename: "scripts.min.js"
  },
  plugins: debug ? [] : [
    new webpack.optimize.DedupePlugin(),
    new webpack.optimize.OccurenceOrderPlugin(),
    new webpack.optimize.UglifyJsPlugin({ mangle: false, sourcemap: false }),
    ],
    module: {
        rules: [
            {
                test: /\.js$/,
                exclude: /node_modules/,
                use: {
                    loader: "babel-loader",
                    options: {
                        presets: ["@babel/preset-env", "@babel/preset-react"]
                    }
                }
            }
        ]
    }
};




to automatically transpile: webpack --watch
